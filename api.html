API Documentation

Complete reference for PinnapleUI v1.0

Window Management

bool Begin(const std::string& title, int width = 1280, int height = 720)

Initialize the window and start a new frame. Call this at the start of your main loop.

if (!PUI::Begin("My App", 1280, 720)) {
    return -1;
}

void End()

End the current frame and swap buffers. Call this at the end of your main loop.

PUI::End();

bool ShouldClose()

Returns true if the window should close (user clicked X).

while (!PUI::ShouldClose()) {
    // Main loop
}

Styling & Configuration

void SetStyle(const Style& style)

Set the global style for the UI.

PUI::Style style = PUI::GetStyle();
style.primaryColor = PUI::Colors::Blue;
PUI::SetStyle(style);

Style& GetStyle()

Get the current style for modification.

bool LoadFont(const std::string& fontPath, float fontSize = 16.0f)

Load a custom TTF font. Call after Begin().

PUI::LoadFont("MyFont.ttf", 20.0f);

void SetUIScale(float scale)

Set UI scale for high DPI displays. 1.0 = 100%, 1.5 = 150%, 2.0 = 200%.

PUI::SetUIScale(1.5f); // 150% scaling

float GetUIScale()

Get the current UI scale.

Widgets

Text

void Text(const std::string& text)

Display text.

PUI::Text("Hello World");

void TextColored(const std::string& text, const Color& color)

Display colored text.

PUI::TextColored("Error!", PUI::Colors::Red);

void Heading(const std::string& text, int level = 1)

Display a heading. Level 1-3 (1 = largest).

PUI::Heading("Title", 1);
PUI::Heading("Subtitle", 2);

Buttons

bool Button(const std::string& label)

Display a button. Returns true when clicked.

if (PUI::Button("Click Me")) {
    // Button was clicked
}

bool ButtonColored(const std::string& label, const Color& color)

Display a colored button.

if (PUI::ButtonColored("Delete", PUI::Colors::Red)) {
    // Delete action
}

Input Fields

bool InputText(const std::string& label, std::string& text, size_t maxLength = 256)

Display a text input field.

std::string username;
PUI::InputText("Username", username);

bool InputPassword(const std::string& label, std::string& text, size_t maxLength = 256)

Display a password input field (shows bullets).

std::string password;
PUI::InputPassword("Password", password);

Checkboxes & Sliders

bool Checkbox(const std::string& label, bool& checked)

Display a checkbox. Returns true when toggled.

bool rememberMe = false;
if (PUI::Checkbox("Remember me", rememberMe)) {
    // Checkbox was toggled
}

bool Slider(const std::string& label, float& value, float min, float max)

Display a slider. Returns true when value changes.

float volume = 50.0f;
if (PUI::Slider("Volume", volume, 0.0f, 100.0f)) {
    // Volume changed
}

void ProgressBar(float progress, float width = 300.0f)

Display a progress bar. Progress is 0.0 to 1.0.

PUI::ProgressBar(0.75f); // 75% complete

Layout

void BeginLayout(LayoutDirection direction, Alignment align)

Start a layout container.

PUI::BeginLayout(PUI::LayoutDirection::Horizontal);
// Widgets here will be laid out horizontally
PUI::EndLayout();

void EndLayout()

End the current layout container.

void Spacing(float amount = -1.0f)

Add vertical spacing. Uses default spacing if amount is -1.

PUI::Spacing(20.0f);

void Separator()

Draw a horizontal separator line.

PUI::Separator();

void SameLine()

Place the next widget on the same line.

PUI::Button("OK");
PUI::SameLine();
PUI::Button("Cancel");

Colors

PinnapleUI provides predefined colors in the PUI::Colors namespace:

Color Usage
PUI::Colors::White Pure white
PUI::Colors::Black Pure black
PUI::Colors::Red Red
PUI::Colors::Green Green
PUI::Colors::Blue Blue
PUI::Colors::Yellow Yellow (#FFD93D)
PUI::Colors::Lime Lime green (#6BCF7F)
PUI::Colors::Pink Pink (#ff6b9d)
PUI::Colors::Gray Gray
PUI::Colors::DarkGray Dark gray
Note: You can create custom colors using Color(r, g, b, a) or Color::FromHex(0xRRGGBB).

Complete Example

#include "PUI.h"

int main() {
    if (!PUI::Begin("My App", 1280, 720)) {
        return -1;
    }
    
    // Optional: Load custom font
    PUI::LoadFont("MyFont.ttf", 20.0f);
    
    // Optional: Set UI scale
    PUI::SetUIScale(1.5f);
    
    std::string username = "";
    bool rememberMe = false;
    float volume = 50.0f;
    
    while (!PUI::ShouldClose()) {
        PUI::Begin("My App", 1280, 720);
        
        PUI::Heading("Welcome!", 1);
        PUI::Text("Enter your details below");
        PUI::Spacing(20.0f);
        
        PUI::InputText("Username", username);
        PUI::Checkbox("Remember me", rememberMe);
        
        if (PUI::Button("Login")) {
            // Handle login
        }
        
        PUI::Separator();
        PUI::Slider("Volume", volume, 0.0f, 100.0f);
        
        PUI::End();
    }
    
    return 0;
}
